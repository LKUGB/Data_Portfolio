def cosine_sim(v1, v2):
    lenV1 = vect_len(v1)
    lenV2 = vect_len(v2)
    innerV1V2 = inner_product(v1, v2)
    cosV1V2 = innerV1V2/(lenV1 * lenV2)
    return cosV1V2
